25int DoublePipeMain(bool debug_flag, bool pu_debug_flag, bool profiling) {
26 DoublePipe double_pipe;
27 NumberPrinter printer;
28 MemoryManager *data_in = new MemoryManager(5, debug_flag);
29 for (size_t it = 0; it < 5; ++it) {
30 Data *holder = new Data(new int(0));
31 data_in->LoadMemoryManager(holder);
32 }
33 Pipeline *pipe = new Pipeline(&printer, data_in, 2, debug_flag, true);
34 pipe->AddProcessingUnit(&double_pipe, 3);
35 pipe->AddProcessingUnit(&printer, 1);
36 pipe->RunPipe();
37
38 u64 t1 = rdtsc();
39 u64 t1c = clock();
40 for (int i = 0; i < 10; ++i) {
41 if (debug_flag) {
42 printf("%s(main) Popping from IN %s\n", LUCID_CYAN, LUCID_NORMAL);
43 }
44 int *data_handler = (int *)data_in->PopFromIn();
45 if (debug_flag) {
46 printf("%s(main) Pushing into OUT %s\n", LUCID_CYAN, LUCID_NORMAL);
47 }
48 data_in->PushIntoOut(data_handler);
49 }
50 pipe->WaitFinish();
51
52 pipe->Profile();
53 return 0;
54}
