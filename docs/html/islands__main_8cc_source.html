26void extraerAlturasDesdeTIFF(const char *archivoTIFF) {
27 TIFF *tif = TIFFOpen(archivoTIFF, "r");
28 if (tif) {
29 uint32_t ancho, alto;
30 TIFFGetField(tif, TIFFTAG_IMAGEWIDTH, &ancho);
31 TIFFGetField(tif, TIFFTAG_IMAGELENGTH, &alto);
32
33 uint32_t tileWidth, tileHeight;
34 TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tileWidth);
35 TIFFGetField(tif, TIFFTAG_TILELENGTH, &tileHeight);
36
37 uint32_t *buffer =
38 (uint32_t *)_TIFFmalloc(tileWidth * tileHeight * sizeof(uint32_t));
39 float **matrizAlturas = new float *[alto];
40 for (uint32_t i = 0; i < alto; ++i) {
41 matrizAlturas[i] = new float[ancho];
42 }
43
44 for (uint32_t y = 0; y < alto; y += tileHeight) {
45 for (uint32_t x = 0; x < ancho; x += tileWidth) {
46 TIFFReadTile(tif, buffer, x, y, 0, 0);
47 for (uint32_t i = 0; i < tileHeight; ++i) {
48 uint32_t fila = y + i;
49 if (fila >= alto) {
50 break;
51 }
52 for (uint32_t j = 0; j < tileWidth; ++j) {
53 uint32_t col = x + j;
54 if (col >= ancho) {
55 break;
56 }
57 matrizAlturas[fila][col] = (float)buffer[i * tileWidth + j];
58 }
59 }
60 }
61 }
62
63 // Hacer algo con la matriz de alturas aquí
64
65 _TIFFfree(buffer);
66 TIFFClose(tif);
67 } else {
68 std::cout << "Error al abrir el archivo TIFF." << std::endl;
69 }
70}

72int IslandsMain(bool debug_flag, bool pu_debug_flag, bool profiling) {
73 TIFF *tif = TIFFOpen("../input_images/ElHierro.tiff", "r");
74 u64 width;
75 u64 height;
76 u64 tile_width;
77 u64 tile_height;
78 u64 unit;
79 TIFFGetField(tif, TIFFTAG_IMAGEWIDTH, &width);
80 TIFFGetField(tif, TIFFTAG_IMAGELENGTH, &height);
81 TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
82 TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_height);
83 TIFFGetField(tif, TIFFTAG_RESOLUTIONUNIT, &unit);
84 printf("%lu", unit);
85
86 u64 *buffer = (u64 *)_TIFFmalloc(tile_width * tile_height * sizeof(u64));
87 float **matrix = new float *[height];
88 for (u64 i = 0; i < height; ++i) {
89 matrix[i] = new float[width];
90 }
91
92 for (u64 y = 0; y < height; y += tile_height) {
93 for (u64 x = 0; x < width; x += tile_width) {
94 TIFFReadTile(tif, buffer, x, y, 0, 0);
95
96 for (u64 i = 0; i < tile_height; ++i) {
97 u64 row = y + i;
98 if (row >= height) break;
99 for (u64 j = 0; j < tile_width; ++j) {
100 u64 col = x + j;
101 if (col > width) break;
102 matrix[row][col] = (double)buffer[i * tile_width + j];
103 printf("%f ", MAT_AT(matrix, row, col));
104 }
105 printf("\n");
106 }
107 }
108 }
109
110 std::string unidadMedida = "AAAA";
111 switch (unit) {
112 case RESUNIT_NONE:
113 unidadMedida = "Sin unidad";
114 break;
115 case RESUNIT_INCH:
116 unidadMedida = "Pulgadas";
117 break;
118 case RESUNIT_CENTIMETER:
119 unidadMedida = "Centímetros";
120 break;
121 }
122
123 _TIFFfree(buffer);
124 TIFFClose(tif);
125 printf("\n%s\n", unidadMedida.c_str());
126 return 0;
127}
