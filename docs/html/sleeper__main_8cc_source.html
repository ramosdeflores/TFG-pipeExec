24int SleeperMain(bool debug_flag, bool pu_debug_flag, bool profiling) {
25 Sleeper sleeper_unit;
26 Sleeper sleeper_unit2;
27 Sleeper sleeper_unit3;
28 NullUnit void_unit;
29 // DATOS
30 int numero_de_datos = 50;
31 // HILOS
32 int numero_de_hilos = numero_de_datos;
33
34 MemoryManager *data_in = new MemoryManager(numero_de_datos, debug_flag);
35 for (int i = 0; i < numero_de_datos; i += 1) {
36 Data *holder = new Data(new int(i));
37 holder->PushExtraData(new Data::DataKey{"profiling", &profiling});
38 data_in->LoadMemoryManager(holder);
39 }
40
41 Pipeline *pipe = new Pipeline(&void_unit, data_in, 1, debug_flag, profiling);
42 pipe->AddProcessingUnit(&sleeper_unit, numero_de_hilos, "d", 1);
43 pipe->AddProcessingUnit(&sleeper_unit2, 3 * numero_de_hilos, "d", 3);
44 pipe->AddProcessingUnit(&sleeper_unit3, numero_de_hilos, "d", 1);
45 pipe->RunPipe();
46
47 for (int i = 0; i < numero_de_datos; ++i) {
48 if (debug_flag) {
49 printf("%s(main) Popping from IN %s\n", LUCID_CYAN, LUCID_NORMAL);
50 }
51 int *data_handler = (int *)data_in->PopFromIn();
52 if (debug_flag) {
53 printf("%s(main) Pushing into OUT %s\n", LUCID_CYAN, LUCID_NORMAL);
54 }
55 data_in->PushIntoOut(data_handler);
56 }
57 pipe->WaitFinish();
58
59 pipe->Profile();
60
61 return 0;
62}
